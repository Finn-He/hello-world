CREATE VIEW booked_group AS 
 SELECT group_booking.group_booking_id,
    group_booking.group_class_id,
    group_booking.user_id,
    group_booking.booking_date,
    group_booking.booking_status,
    group_booking.cancel_date,
    group_booking.attendance_status,
    group_class.group_class_name,
    group_class.group_class_date,
    group_class.group_class_day,
    group_class.group_class_start_time,
    group_class.group_class_end_time,
    group_class.location,
    trainer.first_name,
    trainer.last_name,
    trainer.user_id AS trainer_id
   FROM group_booking
     JOIN group_class ON group_booking.group_class_id = group_class.group_class_id
     JOIN trainer ON group_class.user_id = trainer.user_id;

    pt_class.pt_class_date,

CREATE VIEW class_display AS

    group_class.maximum_number,
    group_class.location,
    group_class.group_class_date,

    trainer.user_id AS trainer_id,
    trainer.first_name AS trainer_name,
    trainer.email_address AS trainer_email,
    trainer.gender
   FROM trainer
     RIGHT JOIN group_class ON group_class.user_id = trainer.user_id;

CREATE VIEW group_attendance AS 
 SELECT booked_group.group_booking_id,
    booked_group.group_class_id,
    booked_group.user_id,
    booked_group.booking_date,
    booked_group.booking_status,
    booked_group.cancel_date,
    booked_group.attendance_status,
    booked_group.group_class_name,
    booked_group.group_class_date,
    booked_group.group_class_day,
    booked_group.group_class_start_time,
    booked_group.group_class_end_time,
    booked_group.location,
    booked_group.first_name,
    booked_group.last_name,
    booked_group.trainer_id,
    date_part('year'::text, to_date(booked_group.group_class_date::text, 'yyyy-mm-dd'::text)) AS class_year,
    date_part('month'::text, to_date(booked_group.group_class_date::text, 'yyyy-mm-dd'::text)) AS class_month
   FROM booked_group
  WHERE booked_group.attendance_status::text = 'present'::text
  ORDER BY booked_group.group_class_date;

CREATE VIEW group_class_attendance AS 
 SELECT group_class.group_class_id,
    group_class.group_class_name,
    group_class.group_class_date,
    date_part('month'::text, to_date(group_class.group_class_date::text, 'yyyy-mm-dd'::text)) AS class_month,
    to_char(to_date(group_class.group_class_date::text, 'yyyy-mm-dd'::text)::timestamp with time zone, 'Mon'::text) AS class_month_name,
    date_part('year'::text, to_date(group_class.group_class_date::text, 'yyyy-mm-dd'::text)) AS class_year,
    count(group_booking.group_class_id) AS count
   FROM group_class
     LEFT JOIN group_booking ON group_booking.group_class_id = group_class.group_class_id
  WHERE group_booking.attendance_status::text = 'present'::text
  GROUP BY group_class.group_class_id;

CREATE VIEW group_class_booking_count AS 
 SELECT group_class.group_class_id,
    count(group_booking.group_class_id) AS count
   FROM group_class
     LEFT JOIN group_booking ON group_booking.group_class_id = group_class.group_class_id AND group_booking.cancel_date IS NULL
  GROUP BY group_class.group_class_id;

CREATE VIEW group_class_view AS 
 SELECT group_class.group_class_name,
    date_part('month'::text, to_date(group_class.group_class_date::text, 'yyyy-mm-dd'::text)) AS class_month,
    date_part('year'::text, to_date(group_class.group_class_date::text, 'yyyy-mm-dd'::text)) AS class_year,
    count(group_booking.group_class_id) AS count
   FROM group_class
     JOIN group_booking ON group_booking.group_class_id = group_class.group_class_id
  WHERE group_booking.attendance_status::text = 'present'::text
  GROUP BY group_class.group_class_name, (date_part('month'::text, to_date(group_class.group_class_date::text, 'yyyy-mm-dd'::text))), (date_part('year'::text, to_date(group_class.group_class_date::text, 'yyyy-mm-dd'::text)));

CREATE VIEW group_classes_view AS
 SELECT group_class.group_class_id,
    group_class.group_class_day,
    to_date(group_class.group_class_date::text, 'YYYY-MM-DD'::text) AS to_date,
    group_class.group_class_start_time,
    group_class.group_class_end_time,
    group_class.group_class_name,
    group_class.location,
    trainer.first_name AS trainer,
    trainer.last_name,
    group_class.maximum_number - (( SELECT group_class_booking_count.count
           FROM group_class_booking_count
          WHERE group_class_booking_count.group_class_id = group_class.group_class_id))
   FROM group_class
     JOIN trainer ON trainer.user_id = group_class.user_id;

CREATE VIEW listmember AS 
 SELECT member.user_id,
    member.first_name,
    member.last_name,
    member.phone_number,
    member.email_address,
    member.address,
    member.date_of_birth,
    member.gender,
    member.goals,
    member.medical_notes,
    member.photo_url,
    membership.membership_type,
    authorisation.user_type,
    authorisation.password
   FROM member
     JOIN authorisation ON authorisation.user_id = member.user_id
     JOIN membership ON membership.user_id = member.user_id;

CREATE VIEW login AS 
 SELECT authorisation.user_id,
    authorisation.password,
    authorisation.user_type,
    member.email_address AS member_email,
    trainer.email_address AS trainer_email,
    manager.email_address AS manager_email
   FROM authorisation
     LEFT JOIN member ON authorisation.user_id = member.user_id
     LEFT JOIN trainer ON authorisation.user_id = trainer.user_id
     LEFT JOIN manager ON authorisation.user_id = manager.user_id;
